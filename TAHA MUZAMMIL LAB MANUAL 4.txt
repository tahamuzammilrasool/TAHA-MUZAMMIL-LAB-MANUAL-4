#include <iostream>
using namespace std;
unsigned long long factorial(unsigned long long X) {
  if (X == 0) {
    return 1;
  } else {
    return X * factorial(X - 1);
  }
}

int main() {
  unsigned long long number;
  cout << "Enter a number to find the factorial of: ";
  cin >> number;

  cout << "The factorial of " << number << " is: " << factorial(number) << endl;

  return 0;
}
